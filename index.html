Connect to Supabase

Enter your Supabase project URL and anon key to enable shared records across all staff devices.

Setup steps:
  1. Go to supabase.com and create a free account
  2. Create a new project
  3. Go to SQL Editor and run the setup SQL below
  4. Go to Settings → API and copy the Project URL and anon key
  5. Paste them above and click Connect
SQL EDITOR — Run this once to set up:
create table public.dogs (
  id bigint generated always as identity primary key,
  created_at timestamptz default now(),
  name text, breed text, ref_no text,
  intake_date text, assessor text,
  score int, traffic_light text,
  saved_at text, data jsonb
);
alter table public.dogs enable row level security;
create policy "Allow all" on public.dogs for all
  using (true) with check (true);

create table public.config (
  key text primary key, value text
);
alter table public.config enable row level security;
create policy "Allow all" on public.config for all
  using (true) with check (true);

insert into public.config (key, value) values
  ('staff_code','NSPCA2024'),
  ('manager_code','NSPCA-MGR')
on conflict (key) do nothing;
